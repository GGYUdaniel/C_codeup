#include <stdio.h>

int main()
{
    long long int a;
    scanf("%lld" ,&a);
    if(a>0)
    {
        printf("plus\n");
        
        if(a%2==0)
        {
            printf("even");
        }
        else
        {
            printf("odd");
        }
    }
    else
    {
        printf("minus\n");
        
        if(a%2==0)
        {
            printf("even");
        }
        else
        {
            printf("odd");
        }
    }
    
    return 0;
}
